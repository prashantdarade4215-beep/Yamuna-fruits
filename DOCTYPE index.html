  <script>

    const products = document.querySelectorAll('.product input');

    const orderDetails = document.getElementById('order-details');

    const totalAmount = document.getElementById('total');

    const sendOrderBtn = document.getElementById('send-order');

 

    products.forEach(input => {

      input.addEventListener('input', updateOrder);

    });

 

    function updateOrder() {

      let total = 0;

      let details = "";

      products.forEach(input => {

        let qty = parseInt(input.value);

        if (qty > 0) {

          let name = input.parentElement.getAttribute('data-name');

          let price = parseInt(input.parentElement.getAttribute('data-price'));

          let subtotal = price * qty;

          total += subtotal;

          details += `${name} - ${qty} × ₹${price} = ₹${subtotal}\n`;

        }

      });

      orderDetails.textContent = details || "No items selected.";

      totalAmount.textContent = total;

    }

 

    sendOrderBtn.addEventListener('click', () => {

      let phoneNumber = "918459905789"; // Your WhatsApp number (with country code, no +)

      let orderText = `Hello, I'd like to order:\n${orderDetails.textContent}\nTotal: ₹${totalAmount.textContent}`;

      window.open(`https://wa.me/${phoneNumber}?text=${encodeURIComponent(orderText)}`, "_blank");

    });

  </script>
